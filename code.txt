    <script>
        document.addEventListener('DOMContentLoaded', function() {
            const heroImageContainer = document.querySelector('.hero-image-container');
            const heroImage = document.querySelector('.hero-image');
            const aboutFloatingImages = document.querySelector('.floating-images');
            const celebrationBg = document.querySelector('.celebration-bg');

            if (heroImage) {
                heroImage.onload = () => {
                    const naturalWidth = heroImage.naturalWidth;
                    const naturalHeight = heroImage.naturalHeight;

                    if (naturalWidth < naturalHeight) {
                        heroImage.classList.add('portrait');
                    } else {
                        heroImage.classList.remove('portrait');
                    }
                };
            }

            document.addEventListener('scroll', function() {
                const scrollY = window.scrollY;

                // Removed parallax effect
                // heroImageContainer.style.transform = `translateY(${scrollY * 0.2}px)`;

                if (aboutFloatingImages) {
                    const scrollYAbout = window.scrollY - aboutFloatingImages.offsetTop + window.innerHeight * 0.5;
                    aboutFloatingImages.querySelectorAll('img').forEach((img, index) => {
                        const speed = (index + 1) * 0.2;
                        img.style.transform = `translateY(${scrollYAbout * speed}px) translateX(${scrollYAbout * speed * 0.1}px)`;
                    });
                }

                if (celebrationBg) {
                    celebrationBg.style.transform = `translateY(${scrollY * 0.1}px)`;
                }
            });

            function createConfetti() {
                const confettiContainer = document.querySelector('.floating-confetti');
                if (!confettiContainer) return;

                const colors = ['#f00', '#0f0', '#00f', '#ff0', '#f0f', '#0ff'];
                const numConfetti = 100;

                for (let i = 0; i < numConfetti; i++) {
                    const confetti = document.createElement('div');
                    confetti.classList.add('confetti');
                    confetti.style.backgroundColor = colors[Math.floor(Math.random() * colors.length)];
                    const size = Math.random() * 10 + 5;
                    confetti.style.width = `${size}px`;
                    confetti.style.height = `${size}px`;
                    confetti.style.left = `${Math.random() * 100}vw`;
                    confetti.style.top = `${Math.random() * 100}vh`;
                    confetti.style.animationDelay = `${Math.random() * 5}s`;
                    confetti.style.animationDuration = `${Math.random() * 5 + 5}s`;
                    confettiContainer.appendChild(confetti);
                }
            }

            createConfetti();

            const floatingImagesContainer = document.querySelector('.floating-images');
            if (floatingImagesContainer) {
                for (let i = 1; i <= 3; i++) {
                    const img = document.createElement('img');
                    img.src = `floating-image${i}.jpg`;
                    img.alt = `Floating ${i}`;
                    img.style.top = `${20 + i * 10}%`;
                    img.style.left = `${10 + i * 15}%`;
                    floatingImagesContainer.appendChild(img);
                }
            }

            const galleryGrid = document.querySelector('.gallery-grid');
            if (galleryGrid) {
                for (let i = 1; i <= 6; i++) {
                    const img = document.createElement('img');
                    img.src = `photo${i}.jpg`;
                    img.alt = `Photo ${i}`;
                    galleryGrid.appendChild(img);
                }
            }

            // Carousel Implementation
            const carouselSlides = document.querySelectorAll('.hero-image-container .carousel-slide');
            const carouselNavItems = document.querySelectorAll('.hero-image-container .carousel-nav-item');
            let currentSlide = 0;
            let intervalId;

            function showSlide(index) {
                carouselSlides.forEach((slide) => {
                    slide.classList.remove('active');
                });
                carouselNavItems.forEach((item) => item.classList.remove('active'));

                carouselSlides[index].classList.add('active');
                carouselNavItems[index].classList.add('active');
                currentSlide = index;
            }

            function nextSlide() {
                let nextIndex = (currentSlide + 1) % carouselSlides.length;
                showSlide(nextIndex);
            }

            function startCarousel() {
                intervalId = setInterval(nextSlide, 5000);
            }

            function stopCarousel() {
                clearInterval(intervalId);
            }

            carouselNavItems.forEach((item, index) => {
                item.addEventListener('click', () => {
                    stopCarousel();
                    showSlide(index);
                    startCarousel();
                });
            });

            heroImageContainer.addEventListener('mouseenter', stopCarousel);
            heroImageContainer.addEventListener('mouseleave', startCarousel);

            showSlide(0);
            startCarousel();
        });

        function createFloatingHearts() {
        const heartContainer = document.getElementById('floatingHearts');

        setInterval(() => {
                const heart = document.createElement('div');
                heart.classList.add('heart');

                heart.style.left = Math.random() * 100 + 'vw';
                heart.style.animationDuration = (Math.random() * 3 + 3) + 's';
                heart.style.transform = `scale(${Math.random() * 0.5 + 0.5}) rotate(-45deg)`;

                heartContainer.appendChild(heart);

                setTimeout(() => {
                    heart.remove();
                }, 6000);
            }, 300);
    }

    createFloatingHearts();

    </script>
